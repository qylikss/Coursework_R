group_by_check <- function(df){ #   
  sale_sum <- aggregate(df, . ~ , FUN=sum) #  
  marja_sum <- aggregate(df,  ~ , FUN=sum) # 
  count_pos <- aggregate(df, . ~ , FUN=sum) #-   
  unique_pos <- summarise(group_by(df, ), 
                          .=length(unique())) #
  client <- summarise(group_by(df, ), =[1]) #t
  date <- summarise(group_by(df, ), .=.[1])
  df_check <- data.frame(date[2],
                         sale_sum[1],
                         client[2],
                         unique_pos[2],
                         count_pos[2],
                         sale_sum[2],
                         marja_sum[2])
  df_check <- df_check[order(df_check$., 
                             decreasing = F), ] 
  return(df_check)
}
#      
mean_int <- function(arr){ 
  summ = 0
  n = length(arr)
  if (n != 1){
    for (i in (1:(n-1))){
      day_diff = as.numeric(gsub(' days', '', 
                                 as.Date(arr[i+1], 
                                         format = "%d/%m/%Y") - 
                                   as.Date(arr[i],
                                           format = "%d/%m/%Y")))
      summ = summ + day_diff
    }
  }
  else{
    summ = 0
  }
  return(signif(summ / n, digits = 4))
}

group_by_client <- function(df_check, for_last_month){
  sale_sum <- aggregate(df_check,  ~ , FUN=sum) #
  loss_sum <- summarise(group_by(df_check, ), =
                          (sum(.) - sum())) #
  count_check <- aggregate(df_check,  ~ , FUN=length) #- 
  first_date <- summarise(group_by(df_check, ), 1=
                            .[1]) # 
  diff_date <- summarise(group_by(df_check, ), = as.numeric(
    gsub(' days', '',
         .[length(.)] - 
           .[1]))) # .  1  
  passed_time <- summarise(group_by(df_check, ), = as.numeric(
    gsub(' days', '',
         for_last_month - 
           .[length(.)]))) # 
  date_int <- summarise(group_by(df_check, ), .=
                          mean_int(.))
  df_client <- data.frame(sale_sum[1],
                          count_check[2],
                          loss_sum[2],
                          sale_sum[2],
                          first_date[2],
                          date_int[2],
                          diff_date[2],
                          passed_time[2]
  )
  return(df_client)
}
